cmake_minimum_required(VERSION 3.5)

project(LoginAdvancedApp VERSION 0.1 LANGUAGES CXX)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(QT NAMES Qt6 Qt5 REQUIRED COMPONENTS Core Gui Network Widgets LinguistTools)
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Core Gui Network Widgets LinguistTools)

set(TS_FILES LoginAdvancedApp_en_150.ts)

set(PROJECT_SOURCES
        authenticationcontroller.h
        authenticationcontroller.cpp
        main.cpp
        mainwindow.cpp
        mainwindow.h
        mainwindow.ui
        loginview.h
        loginview.cpp
        loginapp.h
        loginapp.cpp
        logincontroller.h
        logincontroller.cpp
        ${TS_FILES}
)

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(LoginAdvancedApp
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
    )
# Define target properties for Android with Qt 6 as:
#    set_property(TARGET LoginAdvancedApp APPEND PROPERTY QT_ANDROID_PACKAGE_SOURCE_DIR
#                 ${CMAKE_CURRENT_SOURCE_DIR}/android)
# For more information, see https://doc.qt.io/qt-6/qt-add-executable.html#target-creation

    qt_create_translation(QM_FILES ${CMAKE_SOURCE_DIR} ${TS_FILES})
else()
    if(ANDROID)
        add_library(LoginAdvancedApp SHARED
            ${PROJECT_SOURCES}
        )
# Define properties for Android with Qt 5 after find_package() calls as:
#    set(ANDROID_PACKAGE_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/android")
    else()
        add_executable(LoginAdvancedApp
            ${PROJECT_SOURCES}
        )
    endif()

    qt5_create_translation(QM_FILES ${CMAKE_SOURCE_DIR} ${TS_FILES})
endif()

target_link_libraries(LoginAdvancedApp PRIVATE Qt${QT_VERSION_MAJOR}::Core)
target_link_libraries(LoginAdvancedApp PRIVATE Qt${QT_VERSION_MAJOR}::Gui)
target_link_libraries(LoginAdvancedApp PRIVATE Qt${QT_VERSION_MAJOR}::Network)
target_link_libraries(LoginAdvancedApp PRIVATE Qt${QT_VERSION_MAJOR}::Widgets)
target_link_libraries(LoginAdvancedApp PUBLIC stdc++fs)

set_target_properties(LoginAdvancedApp PROPERTIES
    MACOSX_BUNDLE_GUI_IDENTIFIER my.example.com
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

install(TARGETS LoginAdvancedApp
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

if(QT_VERSION_MAJOR EQUAL 6)
    qt_finalize_executable(LoginAdvancedApp)
endif()
